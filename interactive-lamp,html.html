    <script>

        function toggleLamp() {

            const lamp = document.querySelector('.lamp');

            if (lamp.style.backgroundColor === 'rgb(255, 204, 0)') {

                lamp.style.backgroundColor = 'rgb(255, 69, 0)';

            } else {

                lamp.style.backgroundColor = 'rgb(255, 204, 0)';

            }

        }
